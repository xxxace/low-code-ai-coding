### 2.2 变量/函数命名
| 类型 | 命名风格 | 示例 |

## 四、错误处理方式

### 4.1 标准模式
```typescript
// ✅ 推荐：try-catch + 用户提示
const loadData = async () => {
  loading.value = true
  try {
    const list = await fetchInquiryOrderList({ where: params })
    dataListManager.update(list)
  } catch (err) {
    ElMessageBox.alert(`获取列表错误：${err.message || JSON.stringify(err)}`)
  }
  loading.value = false
}
```

### 4.2 API层错误处理
```typescript
// utils.ts 中的封装
export async function fetchDataList<T>(promiseResponse, option?: FetchWrapperOptions): Promise<T> {
  const errorHandler = function (error: any, type: 'warning' | 'error') {
    if (!option.slient) {
      const message = `${option.errorPrefix || 'error:'}${error.message || JSON.stringify(error)}`
      ElMessage[type](message)
    }
    return option.defaultValue || []
  }
  
  try {
    const res = await promiseResponse
    if (res.statusCode == '1') {
      return res.data as T
    } else {
      return errorHandler(res.message, 'warning')
    }
  } catch (e) {
    return errorHandler(e, 'error')
  }
}
```

### 4.3 错误处理选项
```typescript
type FetchWrapperOptions = {
  slient?: boolean           // 静默模式，不显示错误
  defaultValue?: any         // 错误时返回的默认值
  errorPrefix?: string       // 错误前缀
  messageMode?: 'message' | 'messagebox'  // 提示方式
}
```

return (res.data[0] || option.defaultValue || {}) as T

## 九、明确禁止的写法

### 9.1 变量声明
```typescript
// ❌ 禁止使用 var
var name = 'test'

// ✅ 使用 const/let
const name = 'test'
let count = 0
```

### 9.2 DOM操作
```typescript
// ❌ 禁止直接操作DOM
document.getElementById('xxx').innerHTML = '...'

// ✅ 使用Vue的响应式系统
const content = ref('...')
```

### 9.3 魔法值
```typescript
// ❌ 禁止硬编码魔法值
if (res.statusCode == '1') { ... }
if (diff < 5) { ... }

// ✅ 使用常量
const SUCCESS_CODE = '1'
const MIN_WORKING_DAYS = 5
if (res.statusCode === SUCCESS_CODE) { ... }
```

### 9.4 同步阻塞
```typescript
// ❌ 禁止同步阻塞操作
const data = JSON.parseSync('...')

// ✅ 使用异步操作
const data = await JSON.parse('...')
```

### 9.5 未处理的Promise
```typescript
// ❌ 禁止未处理的Promise
fetchData().then(data => { ... })

// ✅ 使用 async/await 或 catch
try {
  const data = await fetchData()
} catch (e) {
  // 处理错误
}
```

### 9.6 内联样式与类混用
```typescript
// ❌ 避免内联样式
<el-button style="color: #007bff; padding: 0">

// ✅ 使用Tailwind类或scoped样式
<el-button class="!text-blue-500 !p-0">
```
