### 2. 类型安全增强

**问题**：
代码中存在大量 `any` 类型和类型断言，降低类型安全性。

```typescript
// 当前代码 - 类型不安全
const errorHandler = function (error: any, type: 'warning' | 'error') { ... }
const anchorInfo = res.data[0] as unknown as any
const userStore.userInfo as any
```

**风险**：
- 运行时错误风险
- IDE智能提示失效
- 重构困难

**建议改法**：
```typescript
// 定义错误类型
interface ApiError {
  message: string
  code?: string
  statusCode?: string
}

const errorHandler = function (error: ApiError | Error, type: 'warning' | 'error') {
  const message = error.message || '未知错误'
  // ...
}

// 使用类型守卫
function isApiError(error: unknown): error is ApiError {
  return typeof error === 'object' && error !== null && 'message' in error
}

// 定义用户信息类型
interface UserInfo {
  USERNO: string
  username: string
  // ...
}
```

### 3. 常量提取

**问题**：
代码中存在硬编码的魔法值。

```typescript
// 当前代码 - 硬编码
if (res.statusCode == '1') { ... }
if (diff < 5) { ... }
form.STFG = 'BL_D'
form.EXSTFG = 'EX_P'
```

**风险**：
- 难以维护
- 修改时容易遗漏
- 语义不清晰

**建议改法**：
```typescript
// constants/api.ts
export const API_STATUS = {
  SUCCESS: '1',
  ERROR: '0'
} as const

// constants/business.ts
export const BILL_STATUS = {
  DRAFT: 'BL_D',
  SUBMITTED: 'BL_S',
  APPROVED: 'BL_A'
} as const

export const MIN_WORKING_DAYS = 5

// 使用
if (res.statusCode === API_STATUS.SUCCESS) { ... }
if (diff < MIN_WORKING_DAYS) { ... }
form.STFG = BILL_STATUS.DRAFT
```

## 中优先级

### 5. 函数拆分优化

**问题**：
部分函数过长，职责不单一。

```typescript
// 当前代码 - 函数过长（300+行）
const loadData = async () => {
  // 查询参数处理
  // 数据获取
  // 数据转换
  // 图表渲染
  // ...全部在一个函数里
}
```

**风险**：
- 可读性差
- 难以测试
- 难以复用

**建议改法**：
```typescript
// 拆分为多个小函数
const buildQueryParams = () => { ... }
const fetchMainData = async (params) => { ... }
const fetchChartData = async (params) => { ... }
const transformData = (data) => { ... }
const renderChart = (data) => { ... }

const loadData = async () => {
  loading.value = true
  try {
    const params = buildQueryParams()
    const mainData = await fetchMainData(params)
    const chartData = await fetchChartData(params)
    dataListManager.update(transformData(mainData))
    renderChart(chartData)
  } catch (e) {
    handleError(e, '加载数据失败')
  }
  loading.value = false
}
```

### 6. 重复代码提取

**问题**：
`fetchDataList` 和 `fetchDataItem` 存在大量重复代码。

```typescript
// 当前代码 - 重复
export async function fetchDataList<T>(...) {
  const errorHandler = function (error: any, type: 'warning' | 'error') {
    // 重复的错误处理逻辑
  }
  try {
    // ...
  } catch (e) {
    return errorHandler(e, 'error')
  }
}

export async function fetchDataItem<T>(...) {
  const errorHandler = function (error: any, type: 'warning' | 'error') {
    // 完全相同的错误处理逻辑
  }
  // ...
}
```

**风险**：
- 维护成本高
- 修改时容易遗漏

**建议改法**：
```typescript
// 提取公共错误处理
const createErrorHandler = (option: FetchWrapperOptions) => {
  return (error: unknown, type: 'warning' | 'error') => {
    if (!option.slient) {
      const message = `${option.errorPrefix || 'error:'}${getErrorMessage(error)}`
      if (option.messageMode === 'messagebox') {
        ElMessageBox.alert(message, 'Tips', { confirmButtonText: 'OK', type })
      } else {
        ElMessage[type](message)
      }
    }
    return option.defaultValue
  }
}

export async function fetchDataList<T>(promise: Promise<any>, option?: FetchWrapperOptions): Promise<T> {
  const opts = option || {}
  const handleError = createErrorHandler(opts)
  
  try {
    const res = await promise
    return res.statusCode === API_STATUS.SUCCESS ? res.data : handleError(res.message, 'warning')
  } catch (e) {
    return handleError(e, 'error')
  }
}

export async function fetchDataItem<T>(promise: Promise<any>, option?: FetchWrapperOptions): Promise<T> {
  const opts = option || {}
  const handleError = createErrorHandler(opts)
  
  try {
    const res = await promise
    return res.statusCode === API_STATUS.SUCCESS 
      ? (res.data[0] || opts.defaultValue || {}) 
      : handleError(res.message, 'warning')
  } catch (e) {
    return handleError(e, 'error')
  }
}
```

### 8. Loading状态管理

**问题**：
loading状态管理分散，容易遗漏重置。

```typescript
// 当前代码 - 容易遗漏
const loadData = async () => {
  loading.value = true
  try {
    const list = await fetchInquiryOrderList(...)
    dataListManager.update(list)
  } catch (err) {
    ElMessageBox.alert(...)
  }
  loading.value = false  // 如果提前return会遗漏
}
```

**风险**：
- 状态不一致
- UI卡死

**建议改法**：
```typescript
// 使用finally确保重置
const loadData = async () => {
  loading.value = true
  try {
    const list = await fetchInquiryOrderList(...)
    dataListManager.update(list)
  } catch (err) {
    handleError(err, '加载数据失败')
  } finally {
    loading.value = false
  }
}

// 或封装loading管理
const withLoading = async <T>(fn: () => Promise<T>, loadingRef: Ref<boolean>): Promise<T> => {
  loadingRef.value = true
  try {
    return await fn()
  } finally {
    loadingRef.value = false
  }
}

// 使用
const loadData = () => withLoading(async () => {
  const list = await fetchInquiryOrderList(...)
  dataListManager.update(list)
}, loading)
```

### 10. 组件Props类型定义

**问题**：
部分组件props类型定义不够严格。

```typescript
// 当前代码
const props = withDefaults(defineProps<{ data: any[] }>(), {
  data: () => []
})

const tableRef = ref<any>()
```

**风险**：
- 类型检查失效
- 组件使用不规范

**建议改法**：
```typescript
// 定义严格的props类型
interface InquiryOrderTableProps {
  data: InquiryOrderVO[]
  loading?: boolean
  readonly?: boolean
}

const props = withDefaults(defineProps<InquiryOrderTableProps>(), {
  data: () => [],
  loading: false,
  readonly: false
})

// 组件ref类型
const tableRef = ref<InstanceType<typeof ReportTable>>()
```

## 优化实施建议

### 实施顺序
1. **第一阶段（高优先级）**：常量提取
2. **第二阶段（中优先级）**：函数拆分、重复代码提取
3. **第三阶段（低优先级）**：规范化、注释补充

### 注意事项
- 每项优化应独立提交，便于回滚
- 优化前确保有足够的测试覆盖
- 优先处理影响线上稳定性的问题
- 保持向后兼容，避免破坏性修改
